\begin{frame}{Constat de la QA pour Polène}
  \footnotesize Contexte : Polène possède un site e-commerce déployé dans plus de 20 pays. En réalité, il y a 10 sites différents qui doivent être mis en production régulièrement.
  \vskip 8pt
  \begin{columns}[T]
    \column{.49\textwidth}
    \begin{block}{Le quotidien de la QA}
      \footnotesize
      \begin{itemize}
        \item \textbf{Homologation} : tester chaque ticket pour le valider avant la MEP
        \item \textbf{TNR} : vérifier qu'une nouvelle MEP ne casse pas les fonctionnalités déjà existantes
        \item Le tout manuellement sur un environnement de pré-prod, en lien constant avec les devs et la PO
      \end{itemize}
    \end{block}
    \column{.49\textwidth}
    \begin{alertblock}{Ce qui complique la QA sur Polène}
      \footnotesize
      \begin{itemize}
        \item Tickets \textbf{sous-spécifiés} : critères d'acceptance flous
        \item \textbf{Aucune spécification} fiable : seuls les TNR de la page d'accueil
        \item \textbf{Pas d'environnement de test} : pré-prod partagée, Golden Data manquante
      \end{itemize}
    \end{alertblock}
  \end{columns}
  \vskip 16pt
  {\footnotesize $\Rightarrow$~Impossible de réaliser les tests de manière fiable et systématique.}
\end{frame}

\begin{frame}{Documentation des TNR}
  \footnotesize \textbf{Objectif} : pour chaque page du site, documenter les TNR, c'est-à-dire la liste essentielle de chaque élément à vérifier avant les MEP.
  \vskip 8pt
  \begin{alertblock}{Le vrai enjeu : à quel niveau de détail documenter ?}
    \footnotesize
    \begin{itemize}
      \item \textbf{Trop fin} (élément par élément) : la seule page documentée, la Home, l'était ainsi (\alert{28 éléments unitaires}) $\to$ pour un seul QA sur Polène c'est trop compliqué à maintenir
      \item \textbf{Trop vague} : ne sert plus de référence de test fiable
    \end{itemize}
  \end{alertblock}
  \vskip 2pt
  \begin{exampleblock}{Mon choix : documenter par blocs cohérents}
    \footnotesize Chaque composant Shopify = un bloc, niveau calibré par expérimentation sur la Home. Résiste aux changements éditoriaux, reste exploitable.
  \end{exampleblock}
\end{frame}

  \end{table}

    \footnotesize Critère décisif : le \alert{coût de maintenance}. Aucune auto-réparation ne compense

  \end{block}

\begin{frame}{Comment exécuter les tests documentés ?}
  \footnotesize Deux familles : des \textbf{outils dédiés} (BrowserStack, Mr Suricate\dots) où l'on code des scénarios, ou l'\textbf{exploration autonome par IA} où un agent les déduit de la doc.
  \vskip 4pt
  \scriptsize
  \begin{center}
  \begin{tabular}{lll}
    \toprule
    \textbf{Critère} & \textbf{Outil dédié} & \textbf{Exploration IA} \\
    \midrule
    Mise en place   & Élevé (1 j/scénario)     & Modéré (1 prompt) \\
    Maintenance     & Élevée (casse souvent)   & Faible (doc générée) \\
    Réutilisabilité & Limitée                  & Élevée \\
    Déterminisme    & Élevé                    & Modéré (variabilité liée aux LLM) \\
    Vérifiabilité   & Directe (assertions)     & Indirecte (rapport) \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vskip 2pt
  \begin{block}{Choix : Exploration IA via Antigravity}
    \footnotesize Critère décisif : le \alert{coût de maintenance}. Les outils dédiés (Mr Suricate, BrowserStack) réparent un \alert{sélecteur HTML} cassé, mais pas un \alert{changement de contenu} (produit retiré, bannière remplacée), ce qui bouge le plus sur Polène.
  \end{block}
\end{frame}

\begin{frame}{Antigravity : explorer le site comme un "humain"}
  \footnotesize \textbf{Google Antigravity} : un IDE natif IA doté d'un \alert{sous-agent navigateur} piloté par le modèle de vision \alert{Gemini 3 Pro}.
  \vskip 6pt
  \begin{columns}[T]
    \column{.5\textwidth}
    \begin{block}{Comment ça marche ?}
      \footnotesize
      \begin{enumerate}
        \item Un \textbf{prompt} dérivé de la documentation TNR décrit les éléments à vérifier
        \item Le sous-agent \textbf{explore} le site et capture le rendu
        \item Il produit un \textbf{rapport vérifiable} (captures + vidéos)
      \end{enumerate}
      Aucun scénario codé : l'agent les déduit.
    \end{block}
    \column{.46\textwidth}
    \begin{exampleblock}{Pourquoi Gemini pour la vision ?}
      \footnotesize Meilleur modèle de vision en mars (\textit{classement LMArena}). \\
      Il analyse le rendu visuel, pas seulement le DOM.
    \end{exampleblock}
  \end{columns}
  \vskip 16pt
  {\footnotesize $\Rightarrow$~Le coût humain passe de l'écriture de scénarios de tests à la lecture de rapports.}
\end{frame}

\begin{frame}{Obstacle clé : un coût d'exécution intenable TO REWORK}
  % VOIX ~2 min. ETOILE. Ta contribution la + nette : c'est TOI, pas l'IA.
  \begin{columns}[T]
    \column{.5\textwidth}
    \begin{alertblock}{Obstacle}
      \footnotesize Première mesure : $\sim$\alert{1 h} par passe (Home, 9 fonctionnalités).
      Non tenable à l'échelle. Cause : l'agent capturait lui-même en continu.
    \end{alertblock}
    \column{.5\textwidth}
    \begin{examples}
      \footnotesize \textbf{Ré-architecture} : le sous-agent explore \alert{une fois},
      puis \alert{Playwright} pilote les captures. Résultat : $\sim$\alert{20 min} par page.
    \end{examples}
  \end{columns}
  \vskip 6pt
  \begin{block}{Le message à retenir}
    \footnotesize Le difficile n'était pas d'« utiliser l'IA » mais de l'\alert{orchestrer} :
    niveau d'abstraction, architecture du pipeline, coût d'exécution maîtrisé.
  \end{block}
\end{frame}
