\begin{frame}{Reprendre le contrôle en amont : les tickets}
  % VOIX ~2 min. Le point d'intervention remonte : je ne verifie plus seulement, je definis.
  \begin{itemize}
    \item Des QA Checks (critères implicites) vers la \alert{rédaction complète des tickets}
    \item Progression : tickets simples $\to$ \alert{features entières} découpées en plusieurs tickets
    \item Critères rendus explicites en amont $\Rightarrow$ moins de régressions en aval
  \end{itemize}
  \vskip 6pt
  \begin{block}{Le même geste qu'en QA, déplacé en amont}
    \footnotesize Rendre explicite et reproductible ce qui était implicite et porté à la main.
  \end{block}
\end{frame}
